// editor.cs
function ME::enterObjectCreatorMode() {
   unfocus(TedObject);
   focus(MissionEditor);

   MissionObjectList::refresh();
}

function ME::enterTEDMode() {
   unfocus(MissionEditor);
   focus(TedObject);
}

function ME::enterEditMode() {
   if (!isObject(EditorToolsGui)) {
     loadEditorToolWindow(); // For developmental purposes
   }

   GuiNewContentCtrl(MainWindow, SimGui::TSControl);

   postAction(TS_0, Attach, editCamera);
   focus(editCamera);
   cursorOn(MainWindow);
   ME::GetConsoleOptions();

   ME::switchToObjectMode();
}

function ME::enterGameMode(%doRebuild) {
   if(%doRebuild != false)
      ME::RebuildCommandMap();

   unfocus(TedObject);
   unfocus(MissionEditor);
   unfocus(EditCamera);
   loadPlayGui(); //GuiLoadContentCtrl(MainWindow, "play.gui");
   cursorOff(MainWindow);
   //focus(playDelegate);
}
